```jsx
var Person = (function () {
	function Person (name) { // A
		this.name = name;
	}

	Person.prototype.sayHello = function () { // B
		console.log(`Hello`);
	}

	Person.sayHello = function (){ // C
		console.log('Hello');
	}
	
	return Person;
})();
```
